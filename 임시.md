class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        
        vector <int> answer;
        
        vector <int> vec;
        
        for(int i = 1; i <= m; i++) vec.push_back(i);
        
        for (auto q : queries) {
            
            cout << q << endl;
            
            vector<int> :: iterator it = find(vec.begin(), vec.end(), q);
            
            int pos = it - vec.begin();
            
            answer.push_back(pos);
            vec.insert(vec.begin(), *it);
            
            cout << it << endl;
            
            for(auto v : vec) 
                cout << v << endl;
            
            vec.erase(it + 1);
        }
        
        return answer;
    }
};
